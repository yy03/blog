# 1. 顺序表的实现（不可扩容）

# 1.1 自定义API文档

类名 SequenceList
构造方法 SequenceList(int capacity): 创建LinkList对象
成员方法 1. public void clear()：清空线性表
2. public int getLength()：获取链表的长度
3. public boolean isEmpty()：判断链表是否为空
4. public T get(int index)：获取指定位置i的元素（首结点为0）
5. public void insert(T t)：向链表中添加元素
6. public void insert(int index, T t)：向指定位置i处添加元素
7. public T remove(int index)：删除指定位置i的节点，并返回被删除值
8. public int indexOf(T t)：查找元素t在链表中第一次出现的位置,没有返回-1
成员变量 private T[] data：存放数据的数组
private int length：线性表当前长度

# 1.2 代码实现

public class SequenceList<T> implements Iterable{

    private T[] data;
    private int length;

    public SequenceList(int capacity) {

        data = (T[]) new Object[capacity];
        length = 0;
    }

    //    置空线性表
    public void clear() {
        length = 0;
    }

//    判断线性表是否为空
    public boolean isEmpty() {
        return length == 0;
    }

//    获取线性表中元素的个数
    public int getLength() {
        return length;
    }

//    返回线性表中第i个位置的元素
    public T get(int i) {
        return data[i];
    }

//    向线性表中第i个位置前插入数据
    public void insert(int index, T t) {

        for (int i = length - 1; i >= index; i--) {
            data[i+1] = data[i];
        }
        data[index] = t;
        length++;
    }

//    向线性表中添加元素
    public void insert(T t) {
        data[length++] = t;
    }

//    删除并返回线性表中第i个数据
    public T delete(int i) {
        T temp = data[i];
        for (int index = i + 1; index < length; index++) {
            data[index - 1] = data[index];
        }
        length--;
        return temp;
    }

//    返回线性表中元素T第一次出现的下标,未找到返回-1
    public int indexOf(T t) {
        int i =0;
        for (i = 0; i < length; i++)
        {
            if (data[i].equals(t)) {
                return i;
            }
        }
        return -1;
    }

    @Override
    public Iterator iterator() {
        return new myIterator();
    }

    private class myIterator implements Iterator<T> {

        private int cusor;

        public myIterator() {
            this.cusor = 0;
        }

        @Override
        public boolean hasNext() {
            return cusor < length;
        }

        @Override
        public T next() {
            return data[cusor++];
        }
    }
}

# 2. 可扩容顺序表的实现

# 2.1 自定义API文档

类名 SequenceList2
构造方法 SequenceList(): 创建LinkList对象
成员方法 1. public void clear()：清空线性表
2. public int getLength()：获取链表的长度
3. public boolean isEmpty()：判断链表是否为空
4. public T get(int index)：获取指定位置i的元素（首结点为0）
5. public void insert(T t)：向链表中添加元素
6. public void insert(int index, T t)：向指定位置i处添加元素
7. public T remove(int index)：删除指定位置i的节点，并返回被删除值
8. public int indexOf(T t)：查找元素t在链表中第一次出现的位置,没有返回-1
private void reSize(int newSize)：重置数组容量
成员变量 private T[] data：存放数据的数组
private int length：线性表当前长度

# 2.2 代码实现

/*
    容量可变
    只有执行插入或删除操作时对线性表容量进行改变
    扩容时扩大为原数组的2倍
    若当前元素容量不足数组容量的1/4时，将容量变为原数组容量的一半
 */
public class SequenceList2<T> implements Iterable{

    private T[] data;
    private int length;

    public SequenceList2() {

        data = (T[]) new Object[5];
        length = 0;
    }

    //    置空线性表
    public void clear() {
        length = 0;
    }

//    判断线性表是否为空
    public boolean isEmpty() {
        return length == 0;
    }

//    获取线性表中元素的个数
    public int getLength() {
        return length;
    }

//    返回线性表中第i个位置的元素
    public T get(int i) {
        return data[i];
    }

//    向线性表中第i个位置前插入数据
    public void insert(int index, T t) {

        if (length == data.length) {
            reSize(data.length * 2);
        }
        for (int i = length - 1; i >= index; i--) {
            data[i+1] = data[i];
        }
        data[index] = t;
        length++;
    }

//    向线性表中添加元素
    public void insert(T t) {
        if (length == data.length) {
            reSize(data.length * 2);
        }
        data[length++] = t;
    }

//    删除并返回线性表中第i个数据
    public T delete(int i) {
        if (length <= (data.length)/4 ) {
            reSize( (data.length / 2) );
        }
        T temp = data[i];
        for (int index = i + 1; index < length; index++) {
            data[index - 1] = data[index];
        }
        length--;
        return temp;
    }

//    返回线性表中元素T第一次出现的下标,未找到返回-1
    public int indexOf(T t) {
        int i =0;
        for (i = 0; i < length; i++)
        {
            if (data[i].equals(t)) {
                return i;
            }
        }
        return -1;
    }

//    重置数组容量
    private void reSize(int newSize) {
        T[] temp = data;
        data = (T[]) new Object[newSize];
        for (int i = 0; i < length; i++) {
            data[i] = temp[i];
        }
    }

    @Override
    public Iterator iterator() {
        return new myIterator();
    }

    private class myIterator implements Iterator<T> {

        private int cusor;

        public myIterator() {
            this.cusor = 0;
        }

        @Override
        public boolean hasNext() {
            return cusor < length;
        }

        @Override
        public T next() {
            return data[cusor++];
        }
    }
}
